# Dados
n1 = len(ld['PTS'])
xbarra1 = np.mean(ld['PTS'])
S1 = np.var(ld['PTS'],ddof=1) #variancia

n2 = len(lj['PTS'])
xbarra2 = np.mean(lj['PTS'])
S2 = np.var(lj['PTS'],ddof=1) #variancia

alfa = 0.05

print("Vamos testar se as variancias so iguais")
print("H0: S1 - S2 = 0")
print("Ha: S1 - S2 <> 0 \n")
fobs = S1 / S2
pval = (1 - f.cdf(fobs,n1-1,n2-1)) * 2

print("P-valor: " + str ( round( pval , 4 ) ) )
print("Nesse caso, no h evidncias para rejeitar-se a hiptese nula.")
print("Consideramos as varincias iguais. \n\n")

print("Hipteses:")
print("H0: miLD - miLJ <= 0")
print("Ha: miLD - miLJ > 0 \n")

#Varincia combinada ponderada por causa da variancia igual
S2comb = ( (n1-1)*S1 + (n2-1)*S2 ) / ( n1 + n2 - 2)
gl = ( n1 + n2 - 2)

tobs = ( (xbarra1 - xbarra2) - (0) ) / np.sqrt( S2comb/n1 + S2comb/n2 )
pval = t.cdf(tobs,gl) * 2

print("P-valor: " + str ( round( pval , 4 ) ) )
print("Nesse caso, h evidncias para rejeitar-se a hiptese nula.")
